5/* Renders the next frame in a GIF animation to the given position.
6 * You need to call this in order on the same destination for frames
7 * [0..gif->frames_count - 1] to properly render all the frames in the GIF.
8 * The current target bitmap should have the same height as the animation,
9 * and blending should be set to fully copy RGBA.
10 */
11void algif_render_frame(ALGIF_ANIMATION *gif, int frame, int xpos, int ypos) {
12 int x, y, w, h;
13 ALGIF_FRAME *f = &gif->frames[frame];
14 ALGIF_PALETTE *pal;
15 if (frame == 0) {
16 al_draw_filled_rectangle(xpos, ypos, xpos + gif->width,
17 ypos + gif->height, al_map_rgba_f(0, 0, 0, 0));
18 }
19 else {
20 ALGIF_FRAME *p = &gif->frames[frame - 1];
21 if (p->disposal_method == 2) {
22 al_draw_filled_rectangle(xpos + p->xoff, ypos + p->yoff,
23 xpos + p->xoff + p->bitmap_8_bit->w,
24 ypos + p->yoff + p->bitmap_8_bit->h,
25 al_map_rgba_f(0, 0, 0, 0));
26 }
27 else if (p->disposal_method == 3 && gif->store) {
28 al_draw_bitmap_region(gif->store, xpos + p->xoff, ypos + p->yoff,
29 p->bitmap_8_bit->w,
30 p->bitmap_8_bit->h,
31 xpos + p->xoff, ypos + p->yoff, 0);
32 al_destroy_bitmap(gif->store);
33 gif->store = NULL;
34 }
35 }
36 w = f->bitmap_8_bit->w;
37 h = f->bitmap_8_bit->h;
38 if (f->disposal_method == 3) {
39 if (gif->store)
40 al_destroy_bitmap(gif->store);
41 gif->store = al_clone_bitmap(al_get_target_bitmap());
42 }
43 pal = &gif->frames[frame].palette;
44 if (pal->colors_count == 0)
45 pal = &gif->palette;
46
47 for (y = 0; y < h; y++) {
48 for (x = 0; x < w; x++) {
49 int c = f->bitmap_8_bit->data[x + y * f->bitmap_8_bit->w];
50 if (c != f->transparent_index) {
51 al_draw_pixel(xpos + f->xoff + x, ypos + f->yoff + y,
52 al_map_rgb(pal->colors[c].r, pal->colors[c].g,
53 pal->colors[c].b));
54 }
55 }
56 }
57}

59ALGIF_ANIMATION *algif_load_animation_f(ALLEGRO_FILE *file) {
60 ALGIF_ANIMATION *gif = algif_load_raw(file);
61
62 if (!gif)
63 return gif;
64
65 al_init_primitives_addon();
66
67 gif->duration = 0;
68 ALLEGRO_STATE s;
69 al_store_state(&s, ALLEGRO_STATE_TARGET_BITMAP | ALLEGRO_STATE_BLENDER);
70 al_set_blender(ALLEGRO_ADD, ALLEGRO_ONE, ALLEGRO_ZERO);
71 int n = gif->frames_count;
72 int i;
73 for (i = 0; i < n; i++) {
74 ALGIF_FRAME *f = &gif->frames[i];
75 if (i == 0)
76 f->rendered = al_create_bitmap(gif->width, gif->height);
77 else
78 f->rendered = al_clone_bitmap(gif->frames[i - 1].rendered);
79 al_set_target_bitmap(f->rendered);
80 algif_render_frame(gif, i, 0, 0);
81 gif->duration += f->duration;
82 }
83
84 al_restore_state(&s);
85 return gif;
86}
