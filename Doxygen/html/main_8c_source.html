70void MovePlayer(struct Vector2 dir, struct dstr_block* blocks)
71{
72 if (Player != NULL)
73 {
74 //printf("Before Normalize: %lf, %lf\n", dir.x, dir.y);
75 if (dir.x != 0 && dir.y != 0)
76 {
77 float length = sqrt(dir.x * dir.x + dir.y * dir.y);
78 dir.x /= length;
79 dir.y /= length;
80 //dir.x *= 1.3;
81 //dir.y *= 1.3;
82 //printf("After Normalize: %lf, %lf\n", dir.x, dir.y);
83 }
84 struct Vector2 nDir;
85 nDir.x = (dir.x * Player->Speed * deltaTime) * 40;
86 nDir.y = (dir.y * Player->Speed * deltaTime) * 40;
87 float npy = (Player->Transform.position.y + nDir.y);
88 float npx = (Player->Transform.position.x + nDir.x);
89 float halfplayer = ((Player->Transform.scale.x / 2.0) * 128);
90 bool dstr = false;
91 if (!is_on_block(blocks, npx, Player->Transform.position.y, &dstr, Player, bombs, false, 10)) {
92 if (npx >= halfplayer && npx <= ((WIDTH*2)-halfplayer)+1)
93 {
94 if (npx < 0) npx = 0;
95 if (cam_x_offset >= 0)
96 {
97 if ((npx - cam_x_offset - 50 < (Player->Transform.scale.x / 2.0) * 128
98 && nDir.x < 0)
99 ||
100 (npx - cam_x_offset + 50 > WIDTH - ((Player->Transform.scale.x / 2.0) * 128)
101 && nDir.x > 0))
102 {
103 cam_x_offset += nDir.x * 65 * deltaTime;
104 }
105 }
106 Player->Transform.position.x = npx;
107 }
108 }
109 if (!is_on_block(blocks, Player->Transform.position.x, npy, &dstr, Player, bombs, false, 10)) {
110 if (npy > (Player->Transform.scale.y / 2.0) * 128 && npy < HEIGHT + 1 - ((128 * Player->Transform.scale.y)/2.0))
111 {
112 Player->Transform.position.y = npy;
113 }
114 }
115 int cellsizex = (int)(128 * Player->Transform.scale.x);
116 int cellsizey = (int)(128 * Player->Transform.scale.y);
117 float x = (int)(cellsizex * ((int)(Player->Transform.position.x / cellsizex))+cellsizex/2);
118 float y = (int)(cellsizex * ((int)(Player->Transform.position.y / cellsizey)) + cellsizey / 2);
119 if (cam_x_offset > WIDTH)
120 {
121 cam_x_offset = WIDTH;
122 }
123 Player->Transform.gridPosition.x = x;
124 Player->Transform.gridPosition.y = y;
125 if (cam_x_offset < 0) cam_x_offset = 0;
126 }
